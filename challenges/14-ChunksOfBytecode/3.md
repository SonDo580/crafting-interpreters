## Implementation of `malloc` and `free`

https://github.com/SonDo580/k-r/blob/main/notes/8-UnixInterface/7.storage_alloc.c

```c
#define NULL 0

// - if the most restrictive type can be stored at a particular address, all other types may be so
// - assume 'long' is the most restrictive alignment type
typedef long Align;

/* block header */
union header
{
    struct
    {
        union header *ptr; // next block if on free list
        unsigned size;     // size of this block
    } s;
    Align x; // not used; just to force alignment of blocks
};

typedef union header Header;

static Header base;          // empty list to get started
static Header *freep = NULL; // start of free list

/* malloc: general-purpose storage allocator */
void *malloc(unsigned nbytes)
{
    Header *p, *prevp;
    Header *morecore(unsigned); // obtain storage from OS
    unsigned nunits;            // size of allocated block

    // the requested size is rounded up to a proper number of header-sized units.
    // the allocated block contains 1 more unit for the header.
    nunits = (nbytes + sizeof(Header) - 1) / sizeof(Header) + 1;

    if ((prevp = freep) == NULL)
    { // no free list yet
        // create a degenerate free list;
        // it contains 1 block of size 0 and point to itself
        base.s.ptr = freep = prevp = &base;
        base.s.size = 0;
    }

    // begin from freep where the last block was found (initial prevp = freep)
    for (p = prevp->s.ptr;; prevp = p, p = p->s.ptr)
    {
        if (p->s.size >= nunits) // big enough
        {
            if (p->s.size == nunits)     // exactly
                prevp->s.ptr = p->s.ptr; // remove p from free list
            else
            { // too-big
                // allocate tail end (so the original only needs to adjust size)
                p->s.size -= nunits; // adjust size of original block
                p += p->s.size;      // move to allocated block
                p->s.size = nunits;  // size of allocated block
            }

            // set freep to the last found block
            freep = prevp;

            // the returned pointer points at free space (1 unit beyond header)
            return (void *)(p + 1);
        }

        if (p == freep) // wrapped around free list
            if ((p = morecore(nunits)) == NULL)
                return NULL; // none left
    }
}

#define NALLOC 1024 // minimum #units to request

/* morecore: ask system for more memory */
static Header *morecore(unsigned nu)
{
    char *sbrk(int); // UNIX system call sbrk(n) returns a pointer to n more bytes of storage, -1 if there was no space

    char *cp;   // (raw) character pointer
    Header *up; // (header) unit pointer

    // request at least NALLOC units
    if (nu < NALLOC)
        nu = NALLOC;

    cp = sbrk(nu * sizeof(Header));
    if (cp == (char *)-1) // no space found
        return NULL;

    up = (Header *)cp;
    up->s.size = nu;
    free((void *)(up + 1)); // insert additional memory into the arena
    return freep;
}

/* free: put block ap (allocated pointer) in free list */
void free(void *ap)
{
    Header *bp, *p;        // bp: block pointer
    bp = (Header *)ap - 1; // point to block header (ap point to free space)

    // Find the slot for block 'bp' (free list is sorted by memory address)
    for (p = freep; !(bp > p && bp < p->s.ptr); p = p->s.ptr)
        // normal: between 2 existing free blocks (p and p->s.ptr)

        // p > p->s.ptr: wrapped around
        // p == p->s.ptr: single-block free list
        if (p >= p->s.ptr && (bp > p || bp < p->s.ptr))
            break; // freed block at start or end of arena

    if (bp + bp->s.size == p->s.ptr)
    {
        // join to next neighbor
        bp->s.size += p->s.ptr->s.size;
        bp->s.ptr = p->s.ptr->s.ptr;
    }
    else
        bp->s.ptr = p->s.ptr;

    if (p + p->s.size == bp)
    {
        // join to prev neighbor
        p->s.size += bp->s.size;
        p->s.ptr = bp->s.ptr;
    }
    else
        p->s.ptr = bp;

    freep = p; // start looking from p next time
}
```

## Explanation

- Free storage is kept as a list of free blocks
  - Each block contains a size, a pointer to the next block, and the free space itself.
  - The blocks are kept in order of increasing storage address, and the last block points to the first.
- When a request is made (through `malloc`), the free list is scanned until a big-enough block is found (`first fit`)
  - If the block is exactly the size requested, it is unlinked from the free list and returned to user.
  - If the block is too big, it is split, and the proper amount is returned to user (the residue remains on the free list).
  - If no big-enough block is found, request more storage from OS (using `sbrk` system call) and linked to the free list.
- Freeing (`free`) also causes a search of the free list, to find the proper place to insert the block being freed.
  - Handle `fragmentation`: If the block being freed is adjacent to a free block on either side, they are merged into a single block
